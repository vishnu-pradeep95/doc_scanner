```kotlin
class AppPreferences(context: Context) {

    private val prefs: SharedPreferences = context.getSharedPreferences("pdf_scanner_prefs", Context.MODE_PRIVATE)

    companion object {
        const val STYLE_GHIBLI = 0
        const val STYLE_CARTOON = 1
    }

    fun getThemeMode(): Int           // returns prefs.getInt("theme_mode", AppCompatDelegate.MODE_NIGHT_FOLLOW_SYSTEM)
    fun setThemeMode(mode: Int)       // prefs.edit().putInt("theme_mode", mode).apply()
    fun getAppStyle(): Int            // returns prefs.getInt("app_style", STYLE_GHIBLI)
    fun setAppStyle(style: Int)       // prefs.edit().putInt("app_style", style).apply()
    fun getAppStyleName(): String     // when(getAppStyle()) STYLE_CARTOON -> "Cartoon" else -> "Classic"
    fun isCartoonStyle(): Boolean     // getAppStyle() == STYLE_CARTOON
    fun getDefaultFilterIndex(): Int  // returns prefs.getInt("default_filter", 3)  // 3 = Magic
    fun setDefaultFilter(index: Int)  // prefs.edit().putInt("default_filter", index).apply()
    fun getDefaultFilterName(): String // 0->Original, 1->Enhanced, 2->B&W, 3->Magic, 4->Sharpen, else->Magic
    fun getDefaultFilterType(): ImageProcessor.FilterType  // maps index to FilterType enum
}
```
