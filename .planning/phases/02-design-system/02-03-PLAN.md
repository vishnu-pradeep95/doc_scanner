package com.pdfscanner.app.ui

import androidx.annotation.StringRes
import androidx.fragment.app.Fragment
import com.google.android.material.snackbar.Snackbar

/**
 * Show a Snackbar anchored to this fragment's view.
 * Uses fragment's root view — safe even if root is not CoordinatorLayout.
 * Guards against detached fragment (view == null).
 */
fun Fragment.showSnackbar(message: String, duration: Int = Snackbar.LENGTH_SHORT) {
    view?.let { Snackbar.make(it, message, duration).show() }
}

fun Fragment.showSnackbar(@StringRes messageRes: Int, duration: Int = Snackbar.LENGTH_SHORT) {
    view?.let { Snackbar.make(it, messageRes, duration).show() }
}

**Step C — Replace each Toast call:**

Pattern to replace:
  `Toast.makeText(requireContext(), XXXX, Toast.LENGTH_SHORT).show()`
  `Toast.makeText(requireContext(), XXXX, Toast.LENGTH_LONG).show()`
  `Toast.makeText(context, XXXX, Toast.LENGTH_SHORT).show()`
  `Toast.makeText(context, XXXX, Toast.LENGTH_LONG).show()`

Replacement rules:
  - Toast.LENGTH_SHORT → Snackbar.LENGTH_SHORT (default in extension)
  - Toast.LENGTH_LONG → Snackbar.LENGTH_LONG (pass explicitly)
  - R.string.xxx argument → keep as R.string.xxx
  - String literal argument → replace string literal with R.string reference (create new entry if missing)
  - Variable/expression argument (e.g., "Error: ${e.message}") → use String template, call showSnackbar(message)

Replace each call with:
  - `showSnackbar(R.string.xxx)` for string resource args
  - `showSnackbar(R.string.xxx, Snackbar.LENGTH_LONG)` for LENGTH_LONG errors
  - `showSnackbar("Error: ${e.message}")` for dynamic message args

**PdfEditorFragment.kt specific replacements:**
