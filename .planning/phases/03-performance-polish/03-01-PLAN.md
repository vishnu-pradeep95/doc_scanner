**Standard pattern** (fragments with toolbar at top + content + optional bottom bar):
```kotlin
// In onViewCreated(), at the END of the method body:

// Top inset → toolbar
ViewCompat.setOnApplyWindowInsetsListener(binding.toolbar) { v, windowInsets ->
    val insets = windowInsets.getInsets(WindowInsetsCompat.Type.systemBars())
    v.updatePadding(top = insets.top)
    windowInsets   // Do NOT consume — pass insets to siblings

ViewCompat.setOnApplyWindowInsetsListener(binding.buttonsLayout) { v, windowInsets ->

    windowInsets
}
```

Add next to `insertPage()`:

fun insertPages(entries: List<Pair<Int, Uri>>) {
    // entries: list of (originalPosition, uri) sorted ASCENDING by position
    // Insert in ascending order so positions remain correct as list grows
    val current = _pages.value.toMutableList()
    entries.sortedBy { it.first }.forEach { (position, uri) ->
        val clampedPosition = position.coerceIn(0, current.size)
        current.add(clampedPosition, uri)
    }
    _pages.value = current.toList()
}
```

private fun deleteSelectedPagesWithUndo() {
    val pages = viewModel.pages.value ?: return
    val positions = pagesAdapter.getSelectedPositionsInOrder().sortedDescending()
    if (positions.isEmpty()) return

    // Capture the deleted URIs with their original positions BEFORE deletion
    // Positions are sorted descending so we delete from the end first (no index shift)
    val deletedEntries: List<Pair<Int, Uri>> = positions.mapNotNull { pos ->
        if (pos in pages.indices) Pair(pos, pages[pos]) else null
    }

    // Commit all deletions (descending order avoids index shift issues)
    positions.forEach { position ->
        viewModel.removePage(position)
    }
    pagesAdapter.exitSelectionMode()

    val count = deletedEntries.size
    val message = resources.getQuantityString(R.plurals.pages_deleted, count, count)
    Snackbar.make(binding.root, message, Snackbar.LENGTH_LONG)
        .setAction(R.string.undo) {
            // Restore in ascending position order
            viewModel.insertPages(deletedEntries.sortedBy { it.first })
        }
        .show()
}
```
